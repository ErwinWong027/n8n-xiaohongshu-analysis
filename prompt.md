2. AI Agent分析完成后，需要解析对应的数据，并且分别把数据保存到对应的表中，分别有两个表，`authors`表和`notes`表，分别对应作者列表和笔记列表，结构如下：

```sql

-- =========================================
-- 1. authors 表：作者列表
-- =========================================
CREATE TABLE public.authors (
    id              BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  -- Supabase 推荐用 BIGINT
    user_name       TEXT       NOT NULL,
    user_xhs_id     TEXT       NOT NULL UNIQUE,   -- 小红书号，唯一
    subscribers     INTEGER    NOT NULL DEFAULT 0,
    followers       INTEGER    NOT NULL DEFAULT 0,
    likes           INTEGER    NOT NULL DEFAULT 0,
    profile_url     TEXT,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),  -- 如果后续导入旧数据，再 UPDATE
    other_notes     JSONB      NOT NULL DEFAULT '[]',    -- 非 top10Notes 的标题数组
    report          TEXT
);

COMMENT ON COLUMN public.authors.other_notes IS 'allTitles - top10Notes 的余集';
COMMENT ON COLUMN public.authors.report       IS '后续可写入账号分析报告';

-- 索引：常用查询列建议加
CREATE INDEX idx_authors_user_xhs_id ON public.authors (user_xhs_id);
CREATE INDEX idx_authors_created_at  ON public.authors (created_at);


-- =========================================
-- 2. notes 表：笔记列表
-- =========================================
CREATE TABLE public.notes (
    id           BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    author_id    BIGINT      NOT NULL REFERENCES public.authors(id) ON DELETE CASCADE,
    title        TEXT        NOT NULL,
    description  TEXT,
    likes        INTEGER     DEFAULT 0,
    collects     INTEGER     DEFAULT 0,
    is_top10     BOOLEAN     NOT NULL DEFAULT FALSE,
    raw_json     JSONB       NOT NULL,  -- 保留原始字段，方便后续解析
    created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 索引：根据实际查询习惯，可加可不加
CREATE INDEX idx_notes_author_id ON public.notes (author_id);
CREATE INDEX idx_notes_is_top10  ON public.notes (is_top10);
CREATE INDEX idx_notes_created_at ON public.notes (created_at);

```
